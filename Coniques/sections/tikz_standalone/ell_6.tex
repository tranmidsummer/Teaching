\documentclass{standalone}
\usepackage{tikz}
\begin{document}

\begin{tikzpicture}
\def\xC{8}
\def\yC{3}
\def\a{3}
\def\b{2}
\def\ang{0}
% Draw axes
\draw[->] (1,3) -- (15,3) node[below] {$x$};
\draw[->] (8,0) -- (8,6) node[left] {$y$};

% Ellipse
\draw[thick, rotate around={\ang:(\xC,\yC)}] (\xC,\yC) ellipse [x radius=\a, y radius=\b];

% Center
\fill (\xC,\yC) circle (1pt) node[above right] {\footnotesize O};
\node[below right] at (\xC,\yC) {\footnotesize(0,0)};
% Foci
\pgfmathsetmacro{\c}{sqrt(\a*\a - \b*\b)}
\coordinate (F1) at ({\xC+\c*cos(\ang)},{\yC+\c*sin(\ang)});
\coordinate (F2) at ({\xC-\c*cos(\ang)},{\yC-\c*sin(\ang)});
\fill (F2) circle (1pt) node[above] {$F_1$};
\node[below] at (F2) {\footnotesize(-c,0)};
\fill (F1) circle (1pt) node[above] {$F_2$};
\node[below] at (F1) {\footnotesize(c,0)};

% Focus S1
\coordinate (S1) at ({\xC-\a*cos(\ang)},{\yC-\a*sin(\ang)});
\fill (S1) circle (1pt) node[above left] {$S_1$};
\coordinate (S2) at ({\xC+\a*cos(\ang)},{\yC+\a*sin(\ang)});
\fill (S2) circle (1pt) node[above right] {$S_2$};
\coordinate (S3) at ({\xC+\b*sin(\ang)},{\yC+\b*cos(\ang)});
\fill (S3) circle (1pt) node[above right] {$S_3$};
\coordinate (S4) at ({\xC-\b*sin(\ang)},{\yC-\b*cos(\ang)});
\fill (S4) circle (1pt) node[below right] {$S_4$};
\end{tikzpicture}


\end{document}
