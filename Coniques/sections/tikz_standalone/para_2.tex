\documentclass{standalone}
\usepackage{tikz}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\begin{document}


\begin{tikzpicture}

% Parameters
\def\xC{4} % vertex x
\def\yC{1.5} % vertex y
\def\p{1} % focal parameter
\def\ang{30} % rotation angle

% Parabola
\draw[thick, rotate around={\ang:(\xC,\yC)}, domain=-4:3.5, samples=100]
plot ({\xC + (\x*\x)/(4*\p)}, {\yC + \x});

% Vertex
\fill (\xC,\yC) circle (1pt) node[above right] {$S$};

% Focus
\coordinate (F) at ({\xC+\p*cos(\ang)},{\yC+\p*sin(\ang)});
\fill (F) circle (1pt) node[above] {$F$};

% Directrix
\draw[thin, rotate around={\ang:(\xC,\yC)}]
({\xC-\p},{\yC-3}) -- ({\xC-\p},{\yC+2.5});

\node[below left] at ({\xC+2-\p*cos(\ang)},{\yC-2-\p*sin(\ang)}) {$d_F$};

\end{tikzpicture}
\end{document}
