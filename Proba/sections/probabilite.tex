\section{Probabilité}

La probabilité est une \textbf{mesure} de ces évènements.\\
On peut alors définir cette mesure particulière de manière suivante :

\begin{tcolorbox}[colback=cyan!0, colframe=cyan!70!blue, title=Définition d'une loi de probabilité]
\defn{Une \textbf{loi de probabilité $\mathbb{P}$}} est une mesure qui, à chaque évènement $A \subseteq  \Omega$, 
associe un nombre compris entre 0 et 1. 
\begin{equation}
    0\leq \mathbb{P}(A) \leq 1\;
\end{equation}
Avec
\begin{equation}
    \mathbb{P}(\Omega) = 1\;
\end{equation}

Les lois de probabilités étant des mesures, elles doivent alors respecter la définition de la mesure (Définition \ref{def:mesure}). 
\begin{enumerate}
    \item $\mathbb{P}(\emptyset) = 0$.
    \item \textbf{Positivité} : $\mathbb{P}(A) \ge 0$.
    \item \textbf{Additivité dénombrable} : {\color{gray}si $(A_n)_{n \ge 1}$ est une suite d'évènements incompatibles deux à deux, alors
        \[
        \mathbb{P}(A_1 \cup A_2 \cup A_3 \cup \dots) = \mathbb{P}(A_1) + \mathbb{P}(A_2) + \mathbb{P}(A_3) + \dots
    \]
    Cela signifie que la probabilité de l'union des évènements incompatibles est égale à la somme des probabilités de chacun.

    
    \[
        \mathbb{P}\Big(\bigcup_{n=1}^{\infty} A_n\Big) = \sum_{n=1}^{\infty} \mathbb{P}(A_n).
    \]}
En particulier, si $A\subset  \Omega$ et $B \subset \Omega $ sont incompatibles,
\[
\mathbb{P}(A \cup B) = \mathbb{P}(A) + \mathbb{P}(B)
\]
\end{enumerate}

\end{tcolorbox}

\begin{tcolorbox}[colback=cyan!0, colframe=cyan!70!blue, title=Définition de l'équiprobabilité]
\defn{Deux évènements $A$ et $B$ sont équiprobables ssi $\mathbb{P}(A) = \mathbb{P}(B)$.}
\end{tcolorbox}

\begin{tcolorbox}[colback=magenta!0, colframe=red!50!blue, title=Propriétés de l'équiprobabilité]
\defn{Soit une catégorie d'épreuves $\Omega = \{E_1, E_2, \cdots E_n\}$ avec un nombre $n$ fini d'éléments,\\ 
les évènements élémentaires sont équiprobables s'ils ont tous la même probabilité.
\begin{equation}
    \mathbb{P}(E_1) =\mathbb{P}(E_2) = \cdots = \mathbb{P}(E_n) = \frac{1}{n}
\end{equation}
}
\end{tcolorbox}

\begin{tcolorbox}[colback=cyan!0, colframe=cyan!70!blue, title= Définition de la probabilité selon Laplace]
    \defn{La probabilité d'un évènement \(A\) dans un univers \(\Omega\) 
    où tous les évènements élémentaires sont équiprobables est donnée par la formule Laplace :}

\[
P(A) = \frac{\text{nombre de cas favorables}}{\text{nombre de cas possibles}} = \frac{\#A}{\#\Omega}
\]

\end{tcolorbox}
